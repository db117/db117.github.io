# 源码

public static void main(String[] args) {
        String str = "cccc";
        switch (str) {
            case "AaAa":
                System.out.println("AaAa");
                break;
            case "BBBB":
                System.out.println("BBBB");
                break;
            case "AaBB":
                System.out.println("AaBB");
                break;
            default:
                System.out.println(str);
                break;
        }
}

 public static void main(String[] args) {
        String str = "cccc";
        byte var3 = -1;
        switch(str.hashCode()) {
        case 2031744:
            if (str.equals("AaBB")) {
                var3 = 2;
            } else if (str.equals("BBBB")) {
                var3 = 1;
            } else if (str.equals("AaAa")) {
                var3 = 0;
            }
        default:
            switch(var3) {
            case 0:
                System.out.println("AaAa");
                break;
            case 1:
                System.out.println("BBBB");
                break;
            case 2:
                System.out.println("AaBB");
                break;
            default:
                System.out.println(str);
            }

        }
}
