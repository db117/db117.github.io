secret配置tls ​

apiVersion: v1
kind: Secret
metadata:
  name: tls-name
  namespace: kuboard   # 需指定命名空间，一个secret只能在一个命名空间生效
data:
	# 直接把证书，秘钥base64编码放入即可
  tls.crt: Base64crt
  tls.key: Base64key
type: kubernetes.io/tls
